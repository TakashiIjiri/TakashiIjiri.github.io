#sh bashFileName.sh とシェルスクリプトを起動したとき

#スクリプトファイル名を取得
echo $0

#スクリプトファイルのパスを取得
echo $(which $0)

#スクリプトファイルのあるディレクトリを取得
echo $(dirname $(which $0))

#文字の変換 file内のaをbに変換してfileOutに出力
tr a b < file > fileOut

#文字の削除 file内の"\r"を削除してfileOutに出力
tr -d "\r" < file > fileOut
